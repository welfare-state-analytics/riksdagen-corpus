  7def increase_date_precision(date, start=True):
  8	if pd.isna(date):
  9		return date
 10	# Year
 11	if len(date) == 4 and start:
 12		return date + '-01-01'
 13	if len(date) == 4 and not start:
 14		return date + '-12-31'
 15	# Month
 16	if len(date) == 7 and start:
 17		return date + '-01'
 18	if len(date) == 7 and start:
 19		last_day = calendar.monthrange(int(date[0]), int(date[1]))[1]
 20		return date + f'-{last_day}'
 21	# Day
 22	if len(date) == 10:
 23		return date

 26def check_date_overlap(start1, end1, start2, end2):
 27	latest_start = max(start1, start2)
 28	earliest_end = min(end1, end2)
 29	delta = (earliest_end - latest_start).days + 1
 30	overlap = max(0, delta)
 31	return True if overlap > 0 else False

 65def impute_date(db):
 66	db[["start", "end"]] = db[["start", "end"]].astype(str)
 67	db['start'] = db['start'].apply(increase_date_precision, start=True)
 68	db['end'] = db['end'].apply(increase_date_precision, start=False)
 69	db[["start", "end"]] = db[["start", "end"]].apply(pd.to_datetime, format='%Y-%m-%d')
 70	
 71	if 'source' not in db.columns:
 72		return db
 73
 74	# Impute current governments end date
 75	gov_db = pd.read_csv('corpus/metadata/government.csv')
 76	gov_db[["start", "end"]] = gov_db[["start", "end"]].apply(pd.to_datetime, format='%Y-%m-%d')
 77	idx = gov_db['start'].idxmax()
 78	gov_db.loc[idx, 'end'] = gov_db.loc[idx, 'start'] + datetime.timedelta(days = 365*4)
 79
 80	sources = set(db['source'])
 81	if 'member_of_parliament' in sources:
 82		db = impute_member_date(db, gov_db)
 83	if 'minister' in sources:
 84		db = impute_minister_date(db, gov_db)
 85	if 'speaker' in sources:
 86		db = impute_speaker_date(db)
 87	return db

 90def impute_party(db, party):
 91	if 'party' not in db.columns:
 92		db['party'] = pd.Series(dtype=str)
 93	data = []
 94	for i, row in db[db['party'].isnull()].iterrows():	
 95		parties = party[party['wiki_id'] == row['wiki_id']]
 96		if len(set(parties['party'])) == 1:
 97			db.loc[i,'party'] = parties['party'].iloc[0]
 98		if len(set(parties['party'])) >= 2:
 99			for j, sow in parties.iterrows():
100				try:
101					res = check_date_overlap(row['start'], sow['start'], row['end'], sow['end'])
102				except:
103					print("Impute dates on Corpus using impute_date() before imputing parties!\n")
104					raise
105				if res:
106					m = row.copy()
107					m['party'] = sow['party']
108					data.append(m)
109	db = pd.concat([db, pd.DataFrame(data)]).reset_index(drop=True)
110	return db

119def clean_name(db):

128def infer_chamber(db):
129	def _infer_chamber(role):
130		d = {'första': 1, 'andra': 2}
131		match = re.search(r'([a-zåäö]+)\s*(?:kammar)', role)
132		return d[match.group(1)] if match else 0
133	db['chamber'] = db['role'].apply(_infer_chamber).astype(dtype=pd.Int8Dtype())
