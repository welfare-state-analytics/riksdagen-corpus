 12def classify_paragraph(paragraph, classifier, prior=np.log([0.8, 0.2])):
 13    """
 14    Classify paragraph into speeches / descriptions with provided classifier
 15    
 16    Args:
 17        paragraph (str): the text content of a paragraph
 18        classifier (dict): a dictionary that includes 'dim', 'ft' and 'model' keys for classification
 19        prior (np.array): log prior for the classes
 20    """
 21    words = paragraph.split()
 22    V = len(words)
 23    if V == 0:
 24        return prior
 25
 26    x = np.zeros((V, classifier["dim"]))
 27
 28    ft = classifier["ft"]
 29    for ix, word in enumerate(words):
 30        vec = ft.get_word_vector(word)
 31        x[ix] = vec
 32
 33    pred = classifier["model"].predict(x, batch_size=V)
 34    return np.sum(pred, axis=0) + prior

 37def detect_speaker(matched_txt, speaker_db, metadata=None):
 38    """
 39    Detect the speaker of the house
 40
 41    Args:
 42        matched_txt (str): intro text
 43        speaker_db (pd.df): dataframe containing the speaker metadata
 44        metadata (dict): metadata about the protocol. Deprecated.
 45
 46    Returns
 47        speaker_id (str): ID as a string if detected, otherwise None
 48    """
 49    lower_txt = matched_txt.lower()
 50
 51    # Second vice speaker
 52    if re.search('andre vice', lower_txt):
 53        speaker_db = speaker_db[speaker_db["role"].str.contains('andre')]
 54        
 55    # Third vice speaker
 56    elif re.search('tredje vice', lower_txt):
 57        speaker_db = speaker_db[speaker_db["role"].str.contains('tredje')]
 58
 59    # First vice speaker
 60    elif re.search(r'(förste)?\svice', lower_txt):
 61        speaker_db = speaker_db[speaker_db["role"].str.contains('förste')]
 62
 63    # Speaker
 64    elif re.search(r'(herr|fru)?\s?talman', lower_txt):
 65        speaker_db = speaker_db[speaker_db["role"] == 'talman']
 66
 67    if len(set(speaker_db["id"])) == 1:
 68        return speaker_db["id"].iloc[0]

 70def detect_minister(matched_txt, minister_db, intro_dict):
 71    """
 72    Detect a minister
 73
 74    Args:
 75        matched_txt (str): intro text
 76        minister_db (pd.df): dataframe containing the minister metadata
 77        intro_dict (dict): processed information about the intro text, possibly containing eg. 'gender' 
 78    
 79    Returns:
 80        ministed_id (str): ID as a string if detected, otherwise None
 81    """
 82    lower_txt = matched_txt.lower()
 83
 84    # Filter by gender
 85    if 'gender' in intro_dict:
 86        gender = intro_dict["gender"]
 87        minister_db = minister_db[minister_db["gender"] == gender]
 88
 89    # Match by name
 90    if 'name' in intro_dict:
 91        name = intro_dict["name"].lower()
 92        # thage petterson
 93        #print(minister_db)
 94        name_matches = names_in(name, minister_db)
 95        if not name_matches.empty:
 96            if len(set(name_matches["id"])) == 1:
 97                return name_matches["id"].iloc[0]
 98
 99    # Match by role
100    # Catch "utrikesdepartementet"
101    if role := re.search(r'([A-Za-zÀ-ÿ]+)(?:departementet)', lower_txt):
102        r = role.group(0).replace('departementet', '')
103        role_matches = minister_db[minister_db["role"].str.contains(r, regex=False)]
104        if not role_matches.empty:
105            if len(set(role_matches["id"])) == 1:
106                return role_matches["id"].iloc[0]
107
108    # Catch "ministern för utrikes ärendena (...)"
109    elif role := re.search(r'(?:ministern för )([A-Za-zÀ-ÿ]+)', lower_txt):
110        r = role.group(0).split()[-1]
111        role_matches = minister_db[minister_db["role"].str.contains(r, regex=False)]
112        if not role_matches.empty:
113            if len(set(role_matches["id"])) == 1:
114                return role_matches["id"].iloc[0]
115
116    elif role := re.search(r'[A-Za-zÀ-ÿ]+minister', lower_txt):
117        r = role.group(0).replace('minister', '')
118        role_matches = minister_db[minister_db["role"].str.contains(r, regex=False)]
119        if not role_matches.empty:
120            if len(set(role_matches["id"])) == 1:
121                return role_matches["id"].iloc[0]

123def detect_mp(intro_dict, db, party_map=None, match_fuzzily=False):
124    """
125    Match an MP in a text snippet. Returns an MP id (str) if found, otherwise None.
126
127    If multiple people are matched, defaults to returning None.
128    """
129
130    intro_dict["party_abbrev"] = party_map.get(intro_dict.get("party", ""), "")
131    variables = ['party_abbrev', 'specifier', 'name']
132    variables = [v for v in variables if v in list(db.columns)] # removes missing variables
133    variables = sum([list(map(list, combinations(variables, i))) for i in range(len(variables) + 1)], [])[1:]
134    if match_fuzzily:
135        matching_funs = [name_equals, name_almost_equals, names_in]
136    else:
137        matching_funs = [name_equals, names_in]
138    return match_mp(intro_dict, db, variables, matching_funs)

140def intro_to_dict(intro_text, expressions=None):
141    """
142    Convert introduction to a metadata dictionary.
143    Tries to detect 'name', 'gender', 'party', 'specified', as well as some misc things labeled as 'other'.
144
145    Args:
146        intro_text (str): introduction text
147        expressions (list): a list of regular expressions used for metadata detection.
148            Used for optimization, the algorithm works the same if not provided.
149    
150    Returns:
151        d (dict): detected metadata
152    """
153    if expressions is None:
154        expressions = load_expressions(phase="mp")
155    intro_text = intro_text.strip()
156    d = {}
157    for exp, t in expressions:
158        m = exp.search(intro_text)
159        if m is not None:
160            matched_text = m.group(0)
161
162            if t not in d:
163                d[t] = matched_text.strip()
164                intro_text = intro_text.replace(matched_text, " ")
165
166    if "name" in d:
167        s = d["name"]
168        s = s.replace('-', ' ')
169        if ", " in s:
170            s = s.split(", ")
171            s = s[1] + " " + s[0]
172        d["name"] = s
173
174    if "gender" in d:
175        d["gender"] = d["gender"].lower()
176        if d["gender"] in ["herr", "friherre"]:
177            d["gender"] = "man"
178        if d["gender"] in ["fru", "fröken"]:
179            d["gender"] = "woman"
180
181    if "specifier" in d:
182        d["specifier"] = d["specifier"].replace("i ", "")
183        d["specifier"] = d["specifier"].replace("från ", "")
184
185    return d

187def expression_dicts(pattern_db):
188    expressions = dict()
189    manual = dict()
190    for _, row in pattern_db.iterrows():
191        pattern = row["pattern"]
192        exp = re.compile(pattern)
193        # Calculate digest for distringuishing patterns without ugly characters
194        pattern_digest = hashlib.md5(pattern.encode("utf-8")).hexdigest()[:16]
195        expressions[pattern_digest] = exp
196    return expressions, manual

216def combine_intros(elem1, elem2, intro_expressions, other_expressions):
217    """
218    Join intros that have been split as an artifact of the data processing.
219    """
220    if elem1.text is None or elem2.text is None:
221        return False
222    combine = False
223    for exp, _ in other_expressions:
224        for m in exp.finditer(elem1.text.strip()):
225            combine = True
226
227    intro = detect_introduction(elem2.text, intro_expressions)
228    combine = combine and intro is not None
229    combine = combine and "Anf" not in elem2.text
230    if combine:
231        if elem1.text.strip()[-1] == "-":
232            elem2.text = elem1.text.strip()[:-1] + "-" + elem2.text.strip()
233        else:
234            elem2.text = elem1.text + " " + elem2.text
235        elem1.text = ""
236
237    return combine

239def join_text(text1, text2):
240    text1, text2 = list(map(lambda x: ' '.join(x.replace('\n', ' ').split()), [text1, text2]))
241    # Account for words split over textblocks with '-'
242    if text1.endswith('-'):
243        return ''.join([text1[:-1], text2])
244    else:
245        return ' '.join([text1, text2])
